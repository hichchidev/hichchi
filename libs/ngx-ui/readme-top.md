### Button Component

```typescript
import { Component } from '@angular/core';

@Component({
  selector: 'app-example',
  template: `
    <hichchi-button 
      variant="primary" 
      size="large"
      (click)="handleClick()">
      Click Me
    </hichchi-button>

    <hichchi-button 
      variant="secondary" 
      [disabled]="isLoading"
      [loading]="isLoading">
      Submit
    </hichchi-button>
  `,
})
export class ExampleComponent {
  isLoading = false;

  handleClick() {
    console.log('Button clicked!');
  }
}
```
